\begin{frame}[plain]{Problem: Search Space Pruning}
We are given a combinatorial optimization problem with ground set $\mathcal{U}$ and an exact method $\mathcal{OPT}$.
The optimal solution is:
\[X^\star = \operatorname*{arg\,max}_{X \subset \mathcal{U},\, |X| \leq B} f(X)\]

\textbf{Goal:} Find a much smaller set $\mathcal{U}' \subset \mathcal{U}$, with $|\mathcal{U}'| \ll |\mathcal{U}|$, such that solution quality is preserved:
\[
\frac{f\!\left(\mathcal{OPT}(\mathcal{U}')\right)}{f\!\left(\mathcal{OPT}(\mathcal{U})\right)} \geq 1 - \epsilon
\]

\textbf{My role:} Sole first author — formulated the problem, designed the two-stage architecture, implemented all components, and ran all experiments.
\end{frame}

\begin{frame}[plain]{Stage 1: GNN Pre-Pruning}
\begin{itemize}
    \item [\faBullseye] \textbf{Goal:} Eliminate nodes unlikely to belong to the solution before running MCTS.
    \item [\faLightbulb] \textbf{Key Insight:} Nodes in the optimal set tend to share local structural patterns.
    \item [\faBolt] \textbf{Innovation:} Replace handcrafted heuristics with a trainable GNN; initialize node features as $h_{v_i} \sim U(0,1)$, requiring no domain-specific input.
    \item [\faDatabase] \textbf{Training:} Erd\H{o}s-R\'{e}nyi graphs ($n{=}2000$, $p{=}0.1$), budget $= 100$; labels obtained from a heuristic solver.
    \item [\faBalanceScale] \textbf{Loss:} Weighted cross-entropy to handle class imbalance, since solution nodes are rare.
    \item [\faCogs] \textbf{Model:} 2-layer GCN. \textbf{Software:} PyTorch Geometric.
\end{itemize}
\end{frame}

\begin{frame}[plain]{Stage 2: NMCTS Details}
\begin{itemize}
    \item [\faBullseye] \textbf{Goal:} Refine $\mathcal{U}_{\text{GNN}}$ by capturing interactions between nodes that the GNN scores independently.
    \item [\faSitemap] \textbf{Approach:} Neural MCTS guided by the GNN policy; constructs the solution subset one node at a time.
    \item [\faSearch] \textbf{Progressive Widening:} Add a child node if $\lfloor \text{Visits}(N_p)^\alpha \rfloor \geq |\text{Children}(N_p)|$, with $\alpha{=}0.1$.
    \item [\faLevelUpAlt] \textbf{Progressive Deepening:} Search to depth 50; restart from the root when the objective improves by $\geq 1\%$.
    \item [\faChartLine] \textbf{Outcome:} A compact, high-quality pruned set with minimal objective loss.
\end{itemize}
\end{frame}
